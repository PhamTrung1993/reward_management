CREATE TABLE GIFT (
  ID NUMBER PRIMARY KEY,
  CARD_CODE NVARCHAR2(50) NOT NULL,
  USER_RECEIVES NVARCHAR2(9),
  CREATE_TIME DATE,
  USER_TIME DATE,
  STATUS NUMBER(1,0) DEFAULT 0
);
CREATE SEQUENCE GIFT_SEQ
    START WITH 1
    INCREMENT BY 1
    NOCACHE
    NOCYCLE;
    
CREATE TABLE PRIZE_MANAGEMENT (
  ID NUMBER PRIMARY KEY,
  PHONE_NUMBER VARCHAR2(9) NOT NULL,
  EVENT_COUNT NUMBER(2,0)
);
CREATE INDEX idx_PRIZE_MANAGEMENT_PHONE_NUMBER ON PRIZE_MANAGEMENT (PHONE_NUMBER);

CREATE SEQUENCE PRIZE_MANAGEMENT_SEQ
    START WITH 1
    INCREMENT BY 1
    NOCACHE
    NOCYCLE;
